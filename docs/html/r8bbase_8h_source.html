820 inline void calcFIRFilterResponse( const double* flt, int fltlen,
821  const double th, double& re0, double& im0, const int fltlat = 0 )
822 {
823  const double sincr = 2.0 * cos( th );
824  double cvalue1;
825  double svalue1;
826 
827  if( fltlat == 0 )
828  {
829  cvalue1 = 1.0;
830  svalue1 = 0.0;
831  }
832  else
833  {
834  cvalue1 = cos( -fltlat * th );
835  svalue1 = sin( -fltlat * th );
836  }
837 
838  double cvalue2 = cos( -( fltlat + 1 ) * th );
839  double svalue2 = sin( -( fltlat + 1 ) * th );

843

849  fltlen--;

857  svalue2 = tmp;

878 inline void calcFIRFilterResponseAndGroupDelay( const double* const flt,
879  const int fltlen, const double th, double& re, double& im, double& gd )
880 {
881  // Calculate response at "th".
882 
883  calcFIRFilterResponse( flt, fltlen, th, re, im );
884 
885  // Calculate response at close sideband frequencies.
886 
887  const int Count = 2;
888  const double thd2 = 1e-9;
889  double ths[ Count ] = { th - thd2, th + thd2 };
890 
891  if( ths[ 0 ] < 0.0 )
892  {
893  ths[ 0 ] = 0.0;
894  }
895 
896  if( ths[ 1 ] > M_PI )
897  {
898  ths[ 1 ] = M_PI;
899  }
900 
901  double ph1[ Count ];
902  int i;
903 
904  for( i = 0; i < Count; i++ )
905  {
906  double re1;
907  double im1;
908 
909  calcFIRFilterResponse( flt, fltlen, ths[ i ], re1, im1 );
910  ph1[ i ] = atan2( im1, re1 );
911  }
912 
913  if( fabs( ph1[ 1 ] - ph1[ 0 ]) > M_PI )
914  {
915  if( ph1[ 1 ] > ph1[ 0 ])
916  {
917  ph1[ 1 ] -= M_2PI;
918  }
919  else
920  {
921  ph1[ 1 ] += M_2PI;
922  }
923  }
924 
925  const double thd = ths[ 1 ] - ths[ 0 ];
926  gd = ( ph1[ 1 ] - ph1[ 0 ]) / thd;
927 }
